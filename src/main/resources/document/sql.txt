
--  外部ファイルをインポート --
COPY original
	(train_id, name, item_condition_id, category_name, brand_name, price, shipping, item_description)
FROM 
	'train.tsv'
WITH
	DELIMITER E'\t' csv header;

--  テーブルのカラムの型を変更 --
alter table original
alter column price set data type decimal;

--  categoriesテーブルの作成 --
CREATE TABLE categories 
(
	id serial primary key
	, parent_id integer
	, category_name text
	, name_all text
);

--  handmadeカテゴリ > smallcategory > parent_id SQL文 --
UPDATE categories SET parent_Id = 10 WHERE id > 1142;

--  itemsテーブルの作成 --
CREATE TABLE items (
	id serial primary key
	, name text
	, condition integer
	, category integer
	, brand text
	, price decimal
	, shipping integer
	, description text
);

--  itemsテーブルのDB挿入 --
【1.originalでcategory_nameがnullじゃないものを挿入】
insert into items (
	name
	, condition
	, category
	, brand
	, price
	, shipping
	, description
) 
SELECT 
	o.name
	, o.item_condition_id
	, c.id
	, o.brand_name
	, o.price
	, o.shipping
	, o.item_description
FROM original AS o
JOIN categories AS c ON c.name_all = o.category_name;
【2.originalでcategory_nameがnullのものを挿入】
insert into items (
	name
	, condition
	, brand
	, price
	, shipping
	, description
) 
SELECT 
	o.name
	, o.item_condition_id
	, o.brand_name
	, o.price
	, o.shipping
	, o.item_description
FROM original AS o
WHERE o.category_name is NULL;
